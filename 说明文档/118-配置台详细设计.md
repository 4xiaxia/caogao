### 4.1 TypeScript 接口

```typescript
// Provider 数据
interface Provider {
  id: string;
  name: string;
  type: 'openai' | 'anthropic' | 'azure' | 'custom';
  config: {
    baseUrl: string;
    apiKey: string;
    models: string[];
  };
  enabled: boolean;
}

// API Key 数据
interface ApiKey {
```
